SQL Lesson 1: SELECT queries 101

1.Find the title of each film

SELECT title FROM movies;

2.Find the director of each film

 

SELECT director FROM movies;

3.Find the title and director of each film

SELECT title,director FROM movies;

4.Find the title and year of each film

SELECT title,year FROM movies;

5.Find all the information about each film

SELECT * FROM movies;

SQL Lesson 2: Queries with constraints

6.Find the movie with a row id of 6

 

SELECT * FROM movies where id=6

7.Find the movies released in the years between 2000 and 2010

SELECT * FROM movies where year>=2000 and year<=2010

8.Find the movies not released in the years between 2000 and 2010

SELECT * FROM movies where year not between 2000 and 2010

9.Find the first 5 Pixar movies and their release year

SELECT * FROM movies where id<=5

12.Find all the movies (and director) not directed by John Lasseter

13.Find all the WALL-* movies

SQL Lesson 4: Filtering and sorting Query results

14.List all directors of Pixar movies (alphabetically), without duplicates

SELECT distinct director FROM movies order by director;

15.List the last four Pixar movies released (ordered from most recent to least)

SELECT * FROM movies order by year  desc limit 4;

16.List the first five Pixar movies sorted alphabetically

SELECT * FROM movies order by title  asc limit 5;

17.List the next five Pixar movies sorted alphabetically

SELECT * FROM movies order by title asc limit 5 offset 5;

SELECT * FROM north_american_cities where country='Mexico' order by Population desc limit 2

SQL Lesson 6: Multi-table queries with JOINs

23.Find the domestic and international sales for each movie

SELECT * FROM movies m join boxoffice b on m.id = b.movie_id

24.Show the sales numbers for each movie that did better internationally rather than domestically

SELECT * FROM movies m join boxoffice b on m.id = b.movie_id where b.International_sales >Domestic_sales

25.List all the movies by their ratings in descending order

SELECT * FROM movies m join boxoffice b on m.id = b.movie_id order by b.rating desc

SQL Lesson 8: A short note on NULLs

29.Find the name and role of all employees who have not been assigned to a building

SELECT * FROM employees where building is null

30.Find the names of the buildings that hold no employees

SELECT * FROM buildings left join employees on building_name = building where building is null

SQL Lesson 9: Queries with expressions

32.List all movies and their ratings in percent

33.List all movies that were released on even number years

SQL Lesson 10: Queries with aggregates (Pt. 1)

32.Find the longest time that an employee has been at the studio

SELECT max(years_employed) FROM employees;

33.For each role, find the average number of years employed by employees in that role

SELECT role,avg(years_employed)  FROM employees group by role

34.Find the total number of employee years worked in each building

SELECT building, sum(years_employed) FROM employees group by building

SQL Lesson 11: Queries with aggregates (Pt. 2)

35.Find the number of Artists in the studio (without a HAVING clause)

SELECT count(role) FROM employees where role='Artist'

36.Find the number of Employees of each role in the studio

SELECT role, count(role) FROM employees group by role

37.Find the total number of years employed by all Engineers

SELECT sum(years_employed) FROM employees where role='Engineer'

SQL Lesson 12: Order of execution of a Query

38.Find the number of movies each director has directed

SELECT m.Director AS Director, COUNT(*) AS NumMoviesDirected

FROM Movies m GROUP BY m.Director ORDER BY NumMoviesDirected DESC;

39.Find the total domestic and international sales that can be attributed to each director

SELECT director, SUM(domestic_sales + international_sales) as Cumulative_sales_from_all_movies FROM movies  INNER JOIN boxoffice ON movies.id = boxoffice.movie_id GROUP BY director;
